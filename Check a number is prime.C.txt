Check whether a number is prime.

#include<stdio.h>
int main(){
    int i,n,flag=0;
    printf("Input:");
    scanf("%d",&n);
    
    for(i=2;i<n;i++){
     if(n%i!=0){
        continue;
    }
     else{
        flag=1;
        printf("output:Its not a prime number");
        break;
    }
}
    if(flag==0){
        printf(" output:Its a prime number");
    }
    return 0;
}
