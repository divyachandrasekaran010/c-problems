Check Leap Year
Input: 2024
Output: Leap year
Explanation: Leap year if divisible by 4 and not by 100, or divisible by 400.

#include<stdio.h>
int main(){
    int num;
    printf("Input:");
    scanf("%d",&num);
    if(num%4==0 && num%100!=0||num%400==0){
        printf("leap year");
    }
    else
        {
        printf("not leap year");
    }
}
