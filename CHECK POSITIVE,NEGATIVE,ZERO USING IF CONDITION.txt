CHECK POSITIVE ,NEGATIVE,ZERO USING IF CONDITION:


#include<stdio.h>
int main(){
    int a;
    printf("Input:");
    scanf("%d",&a);
    if(a>0){
        printf("positive number");
    }
    else if(a<0){
        printf("Negative number");
    }
    else{
        printf("Zero");
    }
}
