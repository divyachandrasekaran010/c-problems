Write a program to find the LCM of two integers using a for loop.

#include <stdio.h>

int main() {
    int a,b;
    int i;
    printf("input:");
    scanf("%d %d", &a,&b);

    for(i=(a>b?a:b); ;i++) {
        if(i%a==0 && i%b==0){
         printf("output:LCM:%d",i) ;
         break;
        }
    }
    return 0;
}
