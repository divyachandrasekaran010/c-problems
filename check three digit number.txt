Check 3-Digit Number
Input: 345
Output: Three-digit number
Explanation: Number between 100 and 999.

#include<stdio.h>
int main(){
    int num;
    printf("Input:");
    scanf("%d",&num);
    if (num>=100&&num<=999){
        printf("three digit number");
    }
    else{
        printf("the number is not three digit");
    }
    return 0;
}
