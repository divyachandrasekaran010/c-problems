Days in a Month
Input: 2
Output: 28 or 29 days
Explanation: February has 28 or 29 days.
   

#include<stdio.h>
int main(){
    int month;
    printf("input:");
    scanf("%d",&month);
    switch(month){
    case 1:
    case 3:
    case 5:
    case 7:
    case 8:
    case 10:
    case 12:
    printf("31 days");
    break;
    case 2:
    printf("28 or 29 days");
    break;
    case 4:
    case 6:
    case 9:
    case 11:
    printf("30 days");
    default:
    printf("No months has this number ");
    }
    return 0;
}
